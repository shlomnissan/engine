192auto Inverse(const Matrix3& m) {
193 const auto& a = m[0];
194 const auto& b = m[1];
195 const auto& c = m[2];
196
197 auto r0 = Cross(b, c);
198 auto r1 = Cross(c, a);
199 auto r2 = Cross(a, b);
200
201 auto inv_det = 1.0f / Dot(r2, c);
202
203 return Matrix3 {
204 r0.x * inv_det, r0.y * inv_det, r0.z * inv_det,
205 r1.x * inv_det, r1.y * inv_det, r1.z * inv_det,
206 r2.x * inv_det, r2.y * inv_det, r2.z * inv_det
