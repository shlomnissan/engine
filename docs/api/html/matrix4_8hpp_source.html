224[[nodiscard]] inline ENGINE_EXPORT
225auto Inverse(const Matrix4& m) {
226 const auto& a = reinterpret_cast<const Vector3&>(m[0]);
227 const auto& b = reinterpret_cast<const Vector3&>(m[1]);
228 const auto& c = reinterpret_cast<const Vector3&>(m[2]);
229 const auto& d = reinterpret_cast<const Vector3&>(m[3]);
230
231 const float& x = m(3, 0);
232 const float& y = m(3, 1);
233 const float& z = m(3, 2);
234 const float& w = m(3, 3);
235
236 auto s = Cross(a, b);
237 auto t = Cross(c, d);
238 auto u = (a * y) - (b * x);
239 auto v = (c * w) - (d * z);
240
241 const auto inv_det = 1.0f / (Dot(s, v) + Dot(t, u));
242 s *= inv_det;
243 t *= inv_det;
244 u *= inv_det;
245 v *= inv_det;
246
247 auto r0 = Cross(b, v) + t * y;
248 auto r1 = Cross(v, a) - t * x;
249 auto r2 = Cross(d, u) + s * w;
250 auto r3 = Cross(u, c) - s * z;
251
252 return Matrix4 {
253 r0.x, r0.y, r0.z, -Dot(b, t),
254 r1.x, r1.y, r1.z, Dot(a, t),
255 r2.x, r2.y, r2.z, -Dot(d, s),
256 r3.x, r3.y, r3.z, Dot(c, s)
257 };
258}
